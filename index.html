// Clean Architecture — CQRS Handler
public class CreateSceneHandler : IRequestHandler<CreateSceneCommand, SceneDto>
{
  private readonly ISceneRepository _repo;

  public async Task<SceneDto> Handle(CreateSceneCommand cmd, CancellationToken ct)
  {
    var scene = Scene.Create(cmd.Title, cmd.ChapterId);
    await _repo.AddAsync(scene, ct);
    return scene.ToDto();
  }
}
